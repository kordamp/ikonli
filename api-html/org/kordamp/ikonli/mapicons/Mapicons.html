202     public static Mapicons findByDescription(String description) {
203         for (Mapicons font : values()) {
204             if (font.getDescription().equals(description)) {
205                 return font;
206             }
207         }
208         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
209     }

214     Mapicons(String description, char code) {

217     }
