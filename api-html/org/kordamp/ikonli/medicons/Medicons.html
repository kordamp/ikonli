171     public static Medicons findByDescription(String description) {
172         for (Medicons font : values()) {
173             if (font.getDescription().equals(description)) {
174                 return font;
175             }
176         }
177         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
178     }
