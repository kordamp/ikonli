135     public static Microns findByDescription(String description) {
136         for (Microns font : values()) {
137             if (font.getDescription().equals(description)) {
138                 return font;
139             }
140         }
141         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
142     }
