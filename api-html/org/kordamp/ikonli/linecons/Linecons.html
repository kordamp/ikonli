075     private String description;
076     private int code;
077 
078     Linecons(String description, int code) {
079         this.description = description;
080         this.code = code;
081     }
082 
083     public static Linecons findByDescription(String description) {
084         for (Linecons font : values()) {
085             if (font.getDescription().equals(description)) {
086                 return font;
087             }
088         }
089         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
090     }

092     @Override
