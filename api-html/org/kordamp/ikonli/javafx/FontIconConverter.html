18 package org.kordamp.ikonli.javafx;
19 
20 import javafx.css.ParsedValue;
21 import javafx.css.StyleConverter;
22 import javafx.scene.text.Font;
23 import org.kordamp.ikonli.Ikon;
24 
25 /**
26  * @author Andres Almiray
27  */
28 public class FontIconConverter extends StyleConverter<String, Ikon> {
29     private static class Holder {
30         static final FontIconConverter INSTANCE = new FontIconConverter();
31         static final SequenceConverter SEQUENCE_INSTANCE = new SequenceConverter();
32     }
33 
34     public static StyleConverter<String, Ikon> getInstance() {
35         return Holder.INSTANCE;
36     }
37 
38     private FontIconConverter() {
39 
40     }
41 
42     @Override
43     public String toString() {
44         return "FontIconConverter";
45     }
46 
47     @Override
48     public Ikon convert(ParsedValue<String, Ikon> value, Font font) {
49         String description = value.getValue().trim();
50         return IkonResolver.getInstance().resolveIkonHandler(description).resolve(description);
51     }
52 
53     public static final class SequenceConverter extends StyleConverter<String, Ikon[]> {
54         public static SequenceConverter getInstance() {
55             return Holder.SEQUENCE_INSTANCE;
56         }
57 
58         private SequenceConverter() {
59             super();
60         }
61 
62         @Override
63         public Ikon[] convert(ParsedValue<String, Ikon[]> value, Font font) {
64             String[] descriptions = value.getValue().split(",");
65             Ikon[] ikons = new Ikon[descriptions.length];
66             for (int i = 0; i < descriptions.length; i++) {
67                 String description = descriptions[i].trim();
68                 ikons[i] = IkonResolver.getInstance().resolveIkonHandler(description).resolve(description);
69             }
70             return ikons;
71         }
72 
73         @Override
74         public String toString() {
75             return "FontIcon.SequenceConverter";
76         }
77     }
78 }
