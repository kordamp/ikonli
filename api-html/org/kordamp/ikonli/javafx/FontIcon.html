054     public FontIcon() {
055         getStyleClass().setAll("ikonli-font-icon");
056         setIconSize(8);
057         setIconColor(Color.BLACK);
058 
059         fontProperty().addListener((v, o, n) -> {
060             int size = (int) n.getSize();
061             if (size != getIconSize()) {
062                 setIconSize(size);
063             }
064         });
065 
066         fillProperty().addListener((v, o, n) -> {
067             Paint fill = getIconColor();
068             if (!Objects.equals(fill, n)) {
069                 setIconColor(n);
070             }
071         });
072 
073         iconCodeProperty().addListener((v, o, n) -> {
074             if (n != null) {
075                 IkonHandler ikonHandler = IkonResolver.getInstance().resolve(n.getDescription());
076                 setStyle(normalizeStyle(getStyle(), "-fx-font-family", "'" + ikonHandler.getFontFamily() + "'"));
077                 int code = n.getCode();
078                 if (code <= '\uFFFF') {
079                     setText(String.valueOf((char) code));
080                 } else {
081                     char[] charPair = Character.toChars(code);
082                     String symbol = new String(charPair);
083                     setText(symbol);
084                 }
085             }
086         });
087     }

105     @Override
106     public IntegerProperty iconSizeProperty() {
107         if (iconSize == null) {
108             iconSize = new StyleableIntegerProperty(8) {
109                 @Override
110                 public CssMetaData getCssMetaData() {
111                     return StyleableProperties.ICON_SIZE;
112                 }
113 
114                 @Override
115                 public Object getBean() {
116                     return FontIcon.this;
117                 }
118 
119                 @Override
120                 public String getName() {
121                     return "iconSize";
122                 }
123             };
124             iconSize.addListener((v, o, n) -> {
125                 Font font = FontIcon.this.getFont();
126                 if (Math.abs(font.getSize() - n.doubleValue()) >= EPSILON) {
127                     FontIcon.this.setFont(Font.font(font.getFamily(), n.doubleValue()));
128                     FontIcon.this.setStyle(normalizeStyle(getStyle(), "-fx-font-size", n.intValue() + "px"));
129                 }
130             });
131         }
132         return iconSize;
133     }

218     private String normalizeStyle(String style, String key, String value) {
219         int start = style.indexOf(key);
220         if (start != -1) {
221             int end = style.indexOf(";", start);
222             end = end >= start ? end : style.length() - 1;
223             style = style.substring(0, start) + style.substring(end + 1);
224         }
225         return style + key + ": " + value + ";";
226     }

233     public void setIconLiteral(String iconCode) {
234         String[] parts = iconCode.split(":");
235         setIconCode(org.kordamp.ikonli.javafx.IkonResolver.getInstance().resolve(parts[0]).resolve(parts[0]));
236         resolveSize(iconCode, parts);
237         resolvePaint(iconCode, parts);
238     }

245     private void resolveSize(String iconCode, String[] parts) {
246         if (parts.length > 1) {
247             try {
248                 setIconSize(Integer.parseInt(parts[1]));
249             } catch (NumberFormatException e) {
250                 throw invalidDescription(iconCode, e);
251             }
252         }
253     }
254 
255     private void resolvePaint(String iconCode, String[] parts) {
256         if (parts.length > 2) {
257             Paint paint = resolvePaintValue(iconCode, parts[2]);
258             if (paint != null) {
259                 setIconColor(paint);
260             }
261         }
262     }

288     private static Paint resolvePaintValue(String iconCode, String value) {
289         try {
290             return Color.valueOf(value);
291         } catch (IllegalArgumentException e1) {
292             try {
293                 return LinearGradient.valueOf(value);
294             } catch (IllegalArgumentException e2) {
295                 try {
296                     return RadialGradient.valueOf(value);
297                 } catch (IllegalArgumentException e3) {
298                     throw invalidDescription(iconCode, e3);
299                 }
300             }
301         }
302     }

305         throw new IllegalArgumentException("Description " + description + " is not a valid icon description", e);
306     }
