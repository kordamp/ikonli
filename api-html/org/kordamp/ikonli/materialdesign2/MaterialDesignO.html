70     public static MaterialDesignO findByDescription(String description) {
71         for (MaterialDesignO font : values()) {
72             if (font.getDescription().equals(description)) {
73                 return font;
74             }
75         }
76         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
77     }
