113     public static MaterialDesignI findByDescription(String description) {
114         for (MaterialDesignI font : values()) {
115             if (font.getDescription().equals(description)) {
116                 return font;
117             }
118         }
119         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
120     }

122     private String description;
123     private int code;
124 
125     MaterialDesignI(String description, String code) {
126         this.description = description;
127         this.code = Integer.parseInt(code, 16);
128     }

135     @Override
136     public int getCode() {
