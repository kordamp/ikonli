25 public enum MaterialDesignQ implements Ikon {
26     QI("mdi2q-qi", "F0999"),
27     QQCHAT("mdi2q-qqchat", "F0605"),
28     QRCODE("mdi2q-qrcode", "F0432"),
29     QRCODE_EDIT("mdi2q-qrcode-edit", "F08B8"),
30     QRCODE_MINUS("mdi2q-qrcode-minus", "F118C"),
31     QRCODE_PLUS("mdi2q-qrcode-plus", "F118B"),
32     QRCODE_REMOVE("mdi2q-qrcode-remove", "F118D"),
33     QRCODE_SCAN("mdi2q-qrcode-scan", "F0433"),
34     QUADCOPTER("mdi2q-quadcopter", "F0434"),
35     QUALITY_HIGH("mdi2q-quality-high", "F0435"),
36     QUALITY_LOW("mdi2q-quality-low", "F0A0C"),
37     QUALITY_MEDIUM("mdi2q-quality-medium", "F0A0D"),
38     QUORA("mdi2q-quora", "F0D29");
39 
40     public static MaterialDesignQ findByDescription(String description) {
41         for (MaterialDesignQ font : values()) {
42             if (font.getDescription().equals(description)) {
43                 return font;
44             }
45         }
46         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
47     }
48 
49     private String description;
50     private int code;
51 
52     MaterialDesignQ(String description, String code) {
53         this.description = description;
54         this.code = Integer.parseInt(code, 16);
55     }
56 
57     @Override
58     public String getDescription() {
59         return description;
60     }
61 
62     @Override
63     public int getCode() {
64         return code;
65     }
66 }
