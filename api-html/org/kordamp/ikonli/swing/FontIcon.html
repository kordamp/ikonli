018 package org.kordamp.ikonli.swing;
019 
020 import org.kordamp.ikonli.Ikon;
021 import org.kordamp.ikonli.IkonHandler;
022 
023 import javax.swing.Icon;
024 import java.awt.Color;
025 import java.awt.Component;
026 import java.awt.Font;
027 import java.awt.Graphics;
028 import java.awt.Graphics2D;
029 import java.awt.RenderingHints;
030 import java.awt.image.BufferedImage;
031 
032 import static java.awt.GraphicsEnvironment.getLocalGraphicsEnvironment;
033 import static java.util.Objects.requireNonNull;
034 
035 /**
036  * @author Andres Almiray
037  */
038 public class FontIcon implements Icon {
039     private static final Object LOCK = new Object[0];
040 
041     private Font font;
042     private int width = 16;
043     private int height = 16;
044     private BufferedImage buffer;
045 
046     private int iconSize = 16;
047     private Color iconColor = Color.BLACK;
048     private Ikon ikon;
049 
050     public static FontIcon of(Ikon ikon) {
051         return of(ikon, 16, Color.BLACK);
052     }
053 
054     public static FontIcon of(Ikon ikon, int iconSize) {
055         return of(ikon, iconSize, Color.BLACK);
056     }
057 
058     public static FontIcon of(Ikon ikon, Color iconColor) {
059         return of(ikon, 16, iconColor);
060     }
061 
062     public static FontIcon of(Ikon ikon, int iconSize, Color iconColor) {
063         FontIcon icon = new FontIcon();
064         icon.setIkon(ikon);
065         icon.setIconSize(iconSize);
066         icon.setIconColor(iconColor);
067         return icon;
068     }
069 
070     public void paintIcon(Component c, Graphics g, int x, int y) {
071         synchronized (LOCK) {
072             if (buffer == null) {
073                 buffer = new BufferedImage(getIconWidth(), getIconHeight(),
074                     BufferedImage.TYPE_INT_ARGB);
075 
076                 Graphics2D g2 = (Graphics2D) buffer.getGraphics();
077                 g2.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
078                     RenderingHints.VALUE_ANTIALIAS_ON);
079 
080                 g2.setFont(font);
081                 g2.setColor(iconColor);
082 
083                 int sy = g2.getFontMetrics().getAscent();
084                 g2.drawString(String.valueOf(ikon.getCode()), 0, sy);
085 
086                 g2.dispose();
087             }
088 
089             g.drawImage(buffer, x, y, null);
090         }
091     }
092 
093     public Ikon getIkon() {
094         return ikon;
095     }
096 
097     public void setIkon(Ikon ikon) {
098         requireNonNull(iconColor, "Argument 'iconFont' must not be null");
099         this.ikon = ikon;
100         synchronized (LOCK) {
101             IkonHandler ikonHandler = org.kordamp.ikonli.swing.IkonResolver.getInstance().resolve(ikon.getDescription());
102             font = ((Font) ikonHandler.getFont()).deriveFont(Font.PLAIN, iconSize);
103             setProperties();
104         }
105     }
106 
107     public int getIconSize() {
108         return iconSize;
109     }
110 
111     public void setIconSize(int iconSize) {
112         if (iconSize > 0) {
113             this.iconSize = iconSize;
114             if (null != font) {
115                 font = font.deriveFont(Font.PLAIN, iconSize);
116                 setProperties();
117             }
118         }
119     }
120 
121     protected void setProperties() {
122         BufferedImage tmp = new BufferedImage(iconSize, iconSize,
123             BufferedImage.TYPE_INT_ARGB);
124         Graphics2D g2 = getLocalGraphicsEnvironment().createGraphics(tmp);
125         g2.setFont(font);
126         this.width = g2.getFontMetrics().charWidth(ikon.getCode());
127         this.height = g2.getFontMetrics().getHeight();
128 
129         g2.dispose();
130 
131         synchronized (LOCK) {
132             buffer = null;
133         }
134     }
135 
136     public Color getIconColor() {
137         return iconColor;
138     }
139 
140     public void setIconColor(Color iconColor) {
141         requireNonNull(iconColor, "Argument 'iconColor' must not be null");
142         this.iconColor = iconColor;
143         synchronized (LOCK) {
144             buffer = null;
145         }
146     }
147 
148     public int getIconHeight() {
149         return height;
150     }
151 
152     public int getIconWidth() {
153         return width;
154     }
155 }
