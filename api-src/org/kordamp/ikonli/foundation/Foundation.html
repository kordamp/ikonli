343     public static Foundation findByDescription(String description) {
344         for (Foundation font : values()) {
345             if (font.getDescription().equals(description)) {
346                 return font;
347             }
348         }
349         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
350     }

358     }
