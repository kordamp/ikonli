18 /**
19  * @author Andres Almiray

21 public class IkonliIkonResolver extends AbstractIkonHandler {
22     @Override
23     public boolean supports(String description) {
24         return description != null && description.startsWith("ikn-");
25     }
26 
27     @Override
28     public Ikon resolve(String description) {
29         return Ikonli.findByDescription(description);
30     }
31 
32     @Override
33     public String getFontResourcePath() {
34         return "META-INF/resources/ikonli/0.0.0/fonts/ikonli.ttf";
35     }
36 
37     @Override
38     public String getFontFamily() {
39         return "Ikonli";
40     }
41 }
