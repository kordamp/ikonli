016 package org.kordamp.ikonli.javafx;
017 
018 import javafx.beans.property.ObjectProperty;
019 import javafx.beans.property.SimpleObjectProperty;
020 import javafx.beans.value.ChangeListener;
021 import javafx.beans.value.ObservableValue;
022 import javafx.geometry.Pos;
023 import javafx.scene.control.TableCell;
024 import javafx.scene.control.TableColumn;
025 import javafx.util.Callback;
026 import javafx.util.StringConverter;
027 import org.kordamp.ikonli.Ikon;
028 
029 import static java.util.Objects.requireNonNull;
030 
031 public class FontIconTableCell<S, T> extends TableCell<S, T> {
032     private static final String ERROR_CONVERTER_NULL = "Argument 'converter' must not be null";
033 
034     public static <S, T> Callback<TableColumn<S, T>, TableCell<S, T>> forTableColumn() {
035         return param -> new FontIconTableCell<S, T>();
036     }
037 
038     public static <S, T> Callback<TableColumn<S, T>, TableCell<S, T>> forTableColumn(StringConverter<T> converter) {
039         return param -> new FontIconTableCell<S, T>(converter);
040     }
041 
042     private Subscription subscription;
043     private final FontIcon icon;
044     private final ObjectProperty<StringConverter<T>> converter = new SimpleObjectProperty<StringConverter<T>>(this, "converter");
045 
046     @SuppressWarnings("unchecked")
047     public FontIconTableCell() {
048         this(new StringConverter<T>() {
049             @Override
050             public String toString(T object) {
051                 return object != null ? String.valueOf(object) : "";
052             }
053 
054             @Override
055             public T fromString(String string) {
056                 // leave it as is for now
057                 return null;
058             }
059         });
060     }
061 
062     public FontIconTableCell(StringConverter<T> converter) {
063         this.getStyleClass().add("font-icon-table-cell");
064         this.icon = new FontIcon();
065         setConverter(requireNonNull(converter, ERROR_CONVERTER_NULL));
066     }
067 
068     public final ObjectProperty<StringConverter<T>> converterProperty() {
069         return converter;
070     }
071 
072     public final void setConverter(StringConverter<T> converter) {
073         converterProperty().set(requireNonNull(converter, ERROR_CONVERTER_NULL));
074     }
075 
076     public final StringConverter<T> getConverter() {
077         return converterProperty().get();
078     }
079 
080     @Override
081     public void updateItem(T item, boolean empty) {
082         super.updateItem(item, empty);
083 
084         if (empty) {
085             setGraphic(null);
086         } else {
087             if (subscription != null) {
088                 subscription.unsubscribe();
089                 subscription = null;
090             }
091 
092             final TableColumn<S, T> column = getTableColumn();
093             ObservableValue<T> observable = column == null ? null : column.getCellObservableValue(getIndex());
094 
095             if (observable != null) {
096                 ChangeListener<T> listener = (v, o, n) -> setIconCode(n);
097                 observable.addListener(listener);
098                 subscription = () -> observable.removeListener(listener);
099                 setIconCode(observable.getValue());
100             } else if (item != null) {
101                 setIconCode(item);
102             }
103 
104             setGraphic(icon);
105             setAlignment(Pos.CENTER);
106         }
107     }
108 
109     private void setIconCode(T value) {
110         if (value instanceof Ikon) {
111             icon.setIconCode((Ikon) value);
112         } else {
113             icon.setIconLiteral(getConverter().toString(value));
114         }
115     }
116 
117     private interface Subscription {
118         void unsubscribe();
119     }
120 }
