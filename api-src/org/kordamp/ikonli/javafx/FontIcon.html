077  * @author Andres Almiray
078  */
079 public class FontIcon extends Text implements Icon {
080     protected StyleableIntegerProperty iconSize;
081     protected StyleableObjectProperty<Paint> iconColor;
082     private StyleableObjectProperty<Ikon> iconCode;
083 
084     private ChangeListener<Number> iconSizeChangeListener = new ChangeListener<Number>() {
085         @Override
086         public void changed(ObservableValue<? extends Number> v, Number o, Number n) {
087             FontIcon.this.setStyle(getStyle() + "-fx-font-size: " + n + "px;");
088         }
089     };
090     private ChangeListener<Paint> iconColorChangeListener = new ChangeListener<Paint>() {
091         @Override
092         public void changed(ObservableValue<? extends Paint> v, Paint o, Paint n) {
093             FontIcon.this.setFill(n);
094         }
095     };
096     private ChangeListener<Ikon> iconFontChangeListener = new ChangeListener<Ikon>() {
097         @Override
098         public void changed(ObservableValue<? extends Ikon> v, Ikon o, Ikon n) {
099             FontIcon.this.setIconCode(n);
100         }
101     };
102 
103     public FontIcon() {
104         getStyleClass().setAll("font-icon");
105         setIconSize(16);
106         setIconColor(Color.BLACK);
107     }
108 
109     public FontIcon(String iconCode) {
110         this();
111         setIconCode(IkonResolver.getInstance().resolve(iconCode));
112     }
113 
114     public FontIcon(Ikon iconCode) {
115         this();
116         setIconCode(iconCode);
117     }
118 
119     public String toString() {
120         return getIconCode().getDescription() + ":" + getIconSize() + ":" + getIconColor();
121     }
122 
123     @Override
124     public IntegerProperty iconSizeProperty() {
125         if (iconSize == null) {
126             iconSize = new StyleableIntegerProperty(16) {
127                 @Override
128                 public CssMetaData getCssMetaData() {
129                     return StyleableProperties.ICON_SIZE;
130                 }
131 
132                 @Override
133                 public Object getBean() {
134                     return FontIcon.this;
135                 }
136 
137                 @Override
138                 public String getName() {
139                     return "iconSize";
140                 }
141             };
142             iconSize.addListener(iconSizeChangeListener);
143         }
144         return iconSize;
145     }
146 
147     @Override
148     public ObjectProperty<Paint> iconColorProperty() {
149         if (iconColor == null) {
150             iconColor = new StyleableObjectProperty<Paint>(Color.BLACK) {
151                 @Override
152                 public CssMetaData getCssMetaData() {
153                     return StyleableProperties.ICON_COLOR;
154                 }
155 
156                 @Override
157                 public Object getBean() {
158                     return FontIcon.this;
159                 }
160 
161                 @Override
162                 public String getName() {
163                     return "iconColor";
164                 }
165             };
166             iconColor.addListener(iconColorChangeListener);
167         }
168         return iconColor;
169     }
170 
171     public ObjectProperty<Ikon> iconCodeProperty() {
172         if (iconCode == null) {
173             iconCode = new StyleableObjectProperty<Ikon>() {
174                 @Override
175                 public CssMetaData getCssMetaData() {
176                     return StyleableProperties.ICON_CODE;
177                 }
178 
179                 @Override
180                 public Object getBean() {
181                     return FontIcon.this;
182                 }
183 
184                 @Override
185                 public String getName() {
186                     return "iconCode";
187                 }
188             };
189             iconCode.addListener(iconFontChangeListener);
190         }
191         return iconCode;
192     }
193 
194     public ObjectProperty<Ikon> getIconCodeProperty() {
195         return iconCodeProperty();
196     }
197 
198     @Override
199     public void setIconSize(int size) {
200         if (size <= 0) {
201             throw new IllegalStateException("Argument 'size' must be greater than zero.");
202         }
203         iconSizeProperty().set(size);
204     }
205 
206     @Override
207     public int getIconSize() {
208         return iconSizeProperty().get();
209     }
210 
211     @Override
212     public void setIconColor(Paint paint) {
213         requireNonNull(paint, "Argument 'paint' must not be null");
214         iconColorProperty().set(paint);
215     }
216 
217     @Override
218     public Paint getIconColor() {
219         return iconColorProperty().get();
220     }
221 
222     public Ikon getIconCode() {
223         return iconCodeProperty().get();
224     }
225 
226     public void setIconCode(Ikon iconCode) {
227         iconCodeProperty().set(requireNonNull(iconCode, "Argument 'code' must not be null"));
228         setStyle(getStyle() + "-fx-font-family: '" + iconCode.getFontFamily() + "';");
229         setText(String.valueOf(iconCode.getCode()));
230     }
231 
232     public void setIconLiteral(String iconCode) {
233         setIconCode(IkonResolver.getInstance().resolve(iconCode));
234     }
235 
236     public String getIconLiteral() {
237         Ikon ikon = iconCodeProperty().get();
238         return ikon != null ? ikon.getDescription() : null;
239     }

245

249                 }
250 
251                 @Override
252                 public StyleableProperty<Number> getStyleableProperty(FontIcon icon) {

254                 }

256 
257         private static final CssMetaData<FontIcon, Paint> ICON_COLOR =
258             new CssMetaData<FontIcon, Paint>("-fx-icon-color",
259                 PaintConverter.getInstance(), Color.BLACK) {
260 
261                 @Override
262                 public boolean isSettable(FontIcon node) {
263                     return true;
264                 }
265 
266                 @Override
267                 public StyleableProperty<Paint> getStyleableProperty(FontIcon icon) {
268                     return (StyleableProperty<Paint>) icon.iconColorProperty();
269                 }
270             };
271 
272         private static final CssMetaData<FontIcon, Ikon> ICON_CODE =
273             new CssMetaData<FontIcon, Ikon>("-fx-icon-code",
274                 IconFontConverter.getInstance(), null) {
275 
276                 @Override
277                 public boolean isSettable(FontIcon node) {
278                     return true;
279                 }
280 
281                 @Override
282                 public StyleableProperty<Ikon> getStyleableProperty(FontIcon icon) {
283                     return (StyleableProperty<Ikon>) icon.iconCodeProperty();
284                 }
285             };

287         private static final List<CssMetaData<? extends Styleable, ?>> STYLEABLES;
288 
289         static {
290             final List<CssMetaData<? extends Styleable, ?>> styleables =
291                 new ArrayList<CssMetaData<? extends Styleable, ?>>(Text.getClassCssMetaData());
292             styleables.add(ICON_SIZE);
293             styleables.add(ICON_COLOR);
294             styleables.add(ICON_CODE);
295             STYLEABLES = unmodifiableList(styleables);
296         }
297     }
298 
299     public static List<CssMetaData<? extends Styleable, ?>> getClassCssMetaData() {
300         return StyleableProperties.STYLEABLES;
301     }
302 
303     public List<CssMetaData<? extends Styleable, ?>> getCssMetaData() {
304         return FontIcon.getClassCssMetaData();
305     }
