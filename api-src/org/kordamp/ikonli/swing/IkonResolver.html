051 package org.kordamp.ikonli.swing;
052 
053 import org.kordamp.ikonli.IkonHandler;
054 
055 import java.awt.Font;
056 import java.awt.FontFormatException;
057 import java.awt.GraphicsEnvironment;
058 import java.io.IOException;
059 import java.io.InputStream;
060 import java.util.LinkedHashSet;
061 import java.util.ServiceLoader;
062 import java.util.Set;
063 
064 import static java.util.Objects.requireNonNull;
065 
066 /**
067  * @author Andres Almiray
068  */
069 public class IkonResolver {
070     private static final IkonResolver INSTANCE;
071     private static final Set<IkonHandler> HANDLERS = new LinkedHashSet<>();
072 
073     static {
074         INSTANCE = new IkonResolver();
075 
076         ServiceLoader<IkonHandler> loader = ServiceLoader.load(IkonHandler.class);
077         for (IkonHandler handler : loader) {
078             HANDLERS.add(handler);
079 
080             try {
081                 InputStream stream = Thread.currentThread().getContextClassLoader().getResourceAsStream(handler.getFontResourcePath());
082                 Font font = Font.createFont(Font.TRUETYPE_FONT, stream);
083                 GraphicsEnvironment.getLocalGraphicsEnvironment().registerFont(font);
084                 stream.close();
085                 handler.setFont(font);
086             } catch (FontFormatException | IOException ffe) {
087                 throw new IllegalStateException(ffe);
088             }
089         }
090     }
091 
092     private IkonResolver() {
093 
094     }
095 
096     public static IkonResolver getInstance() {
097         return INSTANCE;
098     }
099 
100     public IkonHandler resolve(String value) {
101         requireNonNull(value, "Ikon description must not be null");
102         for (IkonHandler handler : HANDLERS) {
103             if (handler.supports(value)) {
104                 return handler;
105             }
106         }
107         throw new UnsupportedOperationException("Cannot resolve '" + value + "'");
108     }
109 }
