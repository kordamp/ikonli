249     public static Openiconic findByDescription(String description) {
250         for (Openiconic font : values()) {
251             if (font.getDescription().equals(description)) {
252                 return font;
253             }
254         }
255         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
256     }
257 
258     private String description;
259     private char code;
260 
261     Openiconic(String description, char code) {
262         this.description = description;
263         this.code = code;
264     }
265 
266     @Override
267     public String getDescription() {
268         return description;
269     }
270 
271     @Override
272     public char getCode() {
273         return code;
274     }
