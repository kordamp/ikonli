273     public static WeatherIcons findByDescription(String description) {
274         for (WeatherIcons font : values()) {
275             if (font.getDescription().equals(description)) {
276                 return font;
277             }
278         }
279         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
280     }
