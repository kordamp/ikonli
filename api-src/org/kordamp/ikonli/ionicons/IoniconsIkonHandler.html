51 package org.kordamp.ikonli.ionicons;
52 
53 import org.kordamp.ikonli.AbstractIkonHandler;
54 import org.kordamp.ikonli.Ikon;
55 import org.kordamp.ikonli.IkonHandler;
56 import org.kordamp.jipsy.ServiceProviderFor;
57 
58 /**
59  * @author Andres Almiray
60  */
61 @ServiceProviderFor(IkonHandler.class)
62 public class IoniconsIkonHandler extends AbstractIkonHandler {
63     @Override
64     public boolean supports(String description) {
65         return description != null && description.startsWith("ion-");
66     }
67 
68     @Override
69     public Ikon resolve(String description) {
70         return Ionicons.findByDescription(description);
71     }
72 
73     @Override
74     public String getFontResourcePath() {
75         return "META-INF/resources/ionicons/2.0.1/fonts/ionicons.ttf";
76     }
77 
78     @Override
79     public String getFontFamily() {
80         return "Ionicons";
81     }
82 }
