226     public static Websymbols findByDescription(String description) {
227         for (Websymbols font : values()) {
228             if (font.getDescription().equals(description)) {
229                 return font;
230             }
231         }
232         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
233     }
