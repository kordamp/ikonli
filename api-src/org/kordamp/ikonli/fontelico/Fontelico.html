59     public static Fontelico findByDescription(String description) {
60         for (Fontelico font : values()) {
61             if (font.getDescription().equals(description)) {
62                 return font;
63             }
64         }
65         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
66     }

69     private char code;
