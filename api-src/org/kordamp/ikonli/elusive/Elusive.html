325     public static Elusive findByDescription(String description) {
326         for (Elusive font : values()) {
327             if (font.getDescription().equals(description)) {
328                 return font;
329             }
330         }
331         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
332     }
