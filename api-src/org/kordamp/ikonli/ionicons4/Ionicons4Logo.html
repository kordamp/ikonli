090     public static Ionicons4Logo findByDescription(String description) {
091         for (Ionicons4Logo font : values()) {
092             if (font.getDescription().equals(description)) {
093                 return font;
094             }
095         }
096         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
097     }
098 
099     private String description;
100     private char code;
101 
102     Ionicons4Logo(String description, char code) {
103         this.description = description;
104         this.code = code;
105     }
