347     public static Ionicons4IOS findByDescription(String description) {
348         for (Ionicons4IOS font : values()) {
349             if (font.getDescription().equals(description)) {
350                 return font;
351             }
352         }
353         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
354     }
