153     public static BoxiconsLogos findByDescription(String description) {
154         for (BoxiconsLogos font : values()) {
155             if (font.getDescription().equals(description)) {
156                 return font;
157             }
158         }
159         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
160     }
161 
162     private String description;
163     private int code;
164 
165     BoxiconsLogos(String description, int code) {
166         this.description = description;
167         this.code = code;
168     }
169 
170     @Override
171     public String getDescription() {
172         return description;
173     }
174 
175     @Override
176     public int getCode() {
177         return code;
178     }
