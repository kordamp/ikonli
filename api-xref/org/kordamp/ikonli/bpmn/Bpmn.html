134     public static Bpmn findByDescription(String description) {
135         for (Bpmn font : values()) {
136             if (font.getDescription().equals(description)) {
137                 return font;
138             }
139         }
140         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
141     }
142 
143     private String description;
144     private int code;
145 
146     Bpmn(String description, int code) {
147         this.description = description;
148         this.code = code;
149     }
150 
151     @Override
152     public String getDescription() {
153         return description;
154     }
155 
156     @Override
157     public int getCode() {
158         return code;
159     }
