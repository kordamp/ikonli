809     public static FontAwesome findByDescription(String description) {
810         for (FontAwesome font : values()) {
811             if (font.getDescription().equals(description)) {
812                 return font;
813             }
814         }
815         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
816     }
