261     public static MaterialDesignD findByDescription(String description) {
262         for (MaterialDesignD font : values()) {
263             if (font.getDescription().equals(description)) {
264                 return font;
265             }
266         }
267         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
268     }

276     }
