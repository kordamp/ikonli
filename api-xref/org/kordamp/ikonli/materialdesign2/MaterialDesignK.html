86      public static MaterialDesignK findByDescription(String description) {
87          for (MaterialDesignK font : values()) {
88              if (font.getDescription().equals(description)) {
89                  return font;
90              }
91          }
92          throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
93      }
94  
95      private String description;
96      private int code;
97  
98      MaterialDesignK(String description, String code) {
99          this.description = description;
100         this.code = Integer.parseInt(code, 16);
101     }
102 
103     @Override
104     public String getDescription() {

106     }

108     @Override
109     public int getCode() {
