238     public static MaterialDesignR findByDescription(String description) {
239         for (MaterialDesignR font : values()) {
240             if (font.getDescription().equals(description)) {
241                 return font;
242             }
243         }
244         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
245     }

247     private String description;
248     private int code;
249 
250     MaterialDesignR(String description, String code) {
251         this.description = description;
252         this.code = Integer.parseInt(code, 16);
253     }
254 
255     @Override
256     public String getDescription() {
257         return description;
258     }
259 
260     @Override
261     public int getCode() {
262         return code;
263     }
