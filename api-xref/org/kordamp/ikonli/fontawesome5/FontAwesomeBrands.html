453     public static FontAwesomeBrands findByDescription(String description) {
454         for (FontAwesomeBrands font : values()) {
455             if (font.getDescription().equals(description)) {
456                 return font;
457             }
458         }
459         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
460     }
