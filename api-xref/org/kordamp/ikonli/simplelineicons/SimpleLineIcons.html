216     public static SimpleLineIcons findByDescription(String description) {
217         for (SimpleLineIcons font : values()) {
218             if (font.getDescription().equals(description)) {
219                 return font;
220             }
221         }
222         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
223     }
224 
225     private String description;
226     private int code;
227 
228     SimpleLineIcons(String description, int code) {
229         this.description = description;
230         this.code = code;
231     }
232 
233     @Override
234     public String getDescription() {
235         return description;
236     }
237 
238     @Override
239     public int getCode() {
240         return code;
241     }
