310     public static Foundation findByDescription(String description) {
311         for (Foundation font : values()) {
312             if (font.getDescription().equals(description)) {
313                 return font;
314             }
315         }
316         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
317     }
318 
319     private String description;
320     private char code;
321 
322     Foundation(String description, char code) {
323         this.description = description;
324         this.code = code;
325     }
326 
327     @Override
328     public String getDescription() {
329         return description;
330     }
331 
332     @Override
333     public char getCode() {
334         return code;
335     }
