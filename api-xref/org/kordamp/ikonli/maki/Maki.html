90      public static Maki findByDescription(String description) {
91          for (Maki font : values()) {
92              if (font.getDescription().equals(description)) {
93                  return font;
94              }
95          }
96          throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
97      }
