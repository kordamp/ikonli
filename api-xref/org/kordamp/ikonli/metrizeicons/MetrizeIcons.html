328     public static MetrizeIcons findByDescription(String description) {
329         for (MetrizeIcons font : values()) {
330             if (font.getDescription().equals(description)) {
331                 return font;
332             }
333         }
334         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
335     }
336 
337     private String description;
338     private char code;
339 
340     MetrizeIcons(String description, char code) {
341         this.description = description;
342         this.code = code;
343     }
344 
345     @Override
346     public String getDescription() {
347         return description;
348     }
349 
350     @Override
351     public char getCode() {
352         return code;
353     }
