544     public static HawconsFilled findByDescription(String description) {
545         for (HawconsFilled font : values()) {
546             if (font.getDescription().equals(description)) {
547                 return font;
548             }
549         }
550         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
551     }
