313     public static Feather findByDescription(String description) {
314         for (Feather font : values()) {
315             if (font.getDescription().equals(description)) {
316                 return font;
317             }
318         }
319         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
320     }
321 
322     private String description;
323     private int code;
324 
325     Feather(String description, int code) {
326         this.description = description;
327         this.code = code;
328     }
329 
330     @Override
331     public String getDescription() {
332         return description;
333     }
334 
335     @Override
336     public int getCode() {
337         return code;
338     }
