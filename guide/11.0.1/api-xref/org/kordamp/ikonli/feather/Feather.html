158     public static Feather findByDescription(String description) {
159         for (Feather font : values()) {
160             if (font.getDescription().equals(description)) {
161                 return font;
162             }
163         }
164         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
165     }
166 
167     private String description;
168     private char code;
169 
170     Feather(String description, char code) {
171         this.description = description;
172         this.code = code;
173     }
174 
175     @Override
176     public String getDescription() {
177         return description;
178     }
179 
180     @Override
181     public char getCode() {
182         return code;
183     }
