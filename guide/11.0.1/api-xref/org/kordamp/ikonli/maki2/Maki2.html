256     public static Maki2 findByDescription(String description) {
257         for (Maki2 font : values()) {
258             if (font.getDescription().equals(description)) {
259                 return font;
260             }
261         }
262         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
263     }
