344     public static Ionicons4Material findByDescription(String description) {
345         for (Ionicons4Material font : values()) {
346             if (font.getDescription().equals(description)) {
347                 return font;
348             }
349         }
350         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
351     }
