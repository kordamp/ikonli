327     public static Elusive findByDescription(String description) {
328         for (Elusive font : values()) {
329             if (font.getDescription().equals(description)) {
330                 return font;
331             }
332         }
333         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
334     }
335 
336     private String description;
337     private char code;
338 
339     Elusive(String description, char code) {
340         this.description = description;
341         this.code = code;
342     }
343 
344     @Override
345     public String getDescription() {
346         return description;
347     }
348 
349     @Override
350     public char getCode() {
351         return code;
352     }
